USER MODEL
--------------
email: VARCHAR 255 (string)
password: VARCHAR 255 (string)

sequelize model:create --name user --attributes email:string,password:string



PLAYER MODEL
-----------------
first_name: STRING
last_name: STRING
position: STRING
height_feet: INT
height_inches: INT
team_id: INT

sequelize model:create --name player --attributes first_name:string,last_name:string,position:string,height_feet:integer,height_inches:integer,team_id:integer


TEAM MODEL
--------------
city: STRING
conference: STRING 
division: STRING
name: STRING
full_name: STRING
abbreviation: STRING

sequelize model:create --name team --attributes city:string,conference:string,division:string,name:string,full_name:string,abbreviation:string


PLAYERS_TEAMS MODEL
-----------------------
